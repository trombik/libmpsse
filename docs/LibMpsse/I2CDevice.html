def initialize(address:, freq: ClockRates[:one_hundred_khz], device: {})
  @address = address
  @freq = freq
  @device = device
  @repeated_start_required = device.key?(:with_repeated_start) ? device[:with_repeated_start] : true
  @mpsse = new_context
end
